Project Overview: "Sky" + "Link" = "SkyLink"
SkyLink is a sophisticated web application that creates a direct, secure bridge between two users—designated as "Sky" (the controller) and "Link" (the controlled device). When these two instances connect, they form "SkyLink": a powerful peer-to-peer connection enabling real-time media sharing, file transfer, and remote screen control, operating on the same principle as WhatsApp Web but with expanded capabilities for true device interaction.

This project leverages the Temasys Skylink SDK , a mature WebRTC-based library that provides all the underlying infrastructure for peer-to-peer communication, including audio/video streaming, file transfer, and data channels . By building on this SDK, you avoid reinventing complex WebRTC signaling and can focus on creating an intuitive, feature-rich user experience.

// src/services/skylinkService.js
import Skylink from 'skylinkjs';

class SkylinkService {
  constructor() {
    this.skylink = null;
    this.connection = null;
    this.roomListeners = new Map();
    this.localStream = null;
    this.peerId = null;
  }

  initialize(appKey, appSecret, roomName, userData) {
    return new Promise((resolve, reject) => {
      // SkylinkJS initialization with configuration
      this.skylink = new Skylink();
      
      // Set up event handlers
      this.skylink.on('connected', (peerId, peerInfo, isSelf) => {
        if (isSelf) {
          this.peerId = peerId;
          console.log('Connected to room with peer ID:', peerId);
          resolve(peerId);
        }
      });
      
      this.skylink.on('peerJoined', (peerId, peerInfo, isSelf) => {
        if (!isSelf) {
          console.log('Peer joined:', peerId, peerInfo);
          this._notifyListeners('peerJoined', { peerId, peerInfo });
        }
      });
      
      this.skylink.on('peerLeft', (peerId, peerInfo, isSelf) => {
        if (!isSelf) {
          console.log('Peer left:', peerId);
          this._notifyListeners('peerLeft', { peerId, peerInfo });
        }
      });
      
      this.skylink.on('dataTransferStatus', (dataTransferInfo) => {
        this._notifyListeners('fileTransfer', dataTransferInfo);
      });
      
      this.skylink.on('incomingDataTransfer', (dataTransferInfo) => {
        this._notifyListeners('incomingFile', dataTransferInfo);
      });
      
      this.skylink.on('dataTransferData', (data) => {
        this._notifyListeners('message', data);
      });
      
      // Initialize and connect to room
      this.skylink.init({
        appKey: appKey,
        secret: appSecret,
        defaultRoom: {
          name: roomName,
          userData: userData
        }
      });
    });
  }
  
  _notifyListeners(event, data) {
    if (this.roomListeners.has(event)) {
      this.roomListeners.get(event).forEach(callback => callback(data));
    }
  }
  
  addListener(event, callback) {
    if (!this.roomListeners.has(event)) {
      this.roomListeners.set(event, []);
    }
    this.roomListeners.get(event).push(callback);
  }
  
  sendMessage(peerId, message) {
    // Send through data channel
    this.skylink.sendData(peerId, {
      type: 'message',
      content: message,
      timestamp: Date.now()
    });
  }
  
  sendFile(peerId, file) {
    this.skylink.sendFile(peerId, file);
  }
  
  disconnect() {
    if (this.skylink) {
      this.skylink.leaveRoom();
    }
  }
}

export default new SkylinkService();

// src/components/FileTransfer.jsx
import React, { useState, useRef } from 'react';
import './FileTransfer.css';

const FileTransfer = ({ onFileSelect, files, peerConnected }) => {
  const [dragActive, setDragActive] = useState(false);
  const fileInputRef = useRef(null);

  const handleDrag = (e) => {
    e.preventDefault();
    e.stopPropagation();
    if (e.type === "dragenter" || e.type === "dragover") {
      setDragActive(true);
    } else if (e.type === "dragleave") {
      setDragActive(false);
    }
  };

  const handleDrop = (e) => {
    e.preventDefault();
    e.stopPropagation();
    setDragActive(false);
    if (e.dataTransfer.files && e.dataTransfer.files[0]) {
      handleFiles(e.dataTransfer.files);
    }
  };

  const handleFiles = (fileList) => {
    Array.from(fileList).forEach(file => {
      onFileSelect(file);
    });
  };

  const handleButtonClick = () => {
    fileInputRef.current.click();
  };

  const getFileIcon = (filename) => {
    const ext = filename.split('.').pop().toLowerCase();
    if (['jpg', 'jpeg', 'png', 'gif'].includes(ext)) return '🖼️';
    if (['mp4', 'mov', 'avi'].includes(ext)) return '🎬';
    if (['mp3', 'wav'].includes(ext)) return '🎵';
    if (['pdf'].includes(ext)) return '📄';
    if (['doc', 'docx'].includes(ext)) return '📝';
    if (['zip', 'rar'].includes(ext)) return '🗜️';
    return '📁';
  };

  return (
    <div className="file-transfer">
      <div 
        className={`drop-zone ${dragActive ? 'active' : ''} ${!peerConnected ? 'disabled' : ''}`}
        onDragEnter={handleDrag}
        onDragLeave={handleDrag}
        onDragOver={handleDrag}
        onDrop={handleDrop}
        onClick={handleButtonClick}
      >
        <input
          ref={fileInputRef}
          type="file"
          multiple
          onChange={(e) => handleFiles(e.target.files)}
          style={{ display: 'none' }}
          disabled={!peerConnected}
        />
        <div className="drop-zone-content">
          <span className="icon">📤</span>
          <p>Drag & drop files here or click to select</p>
          {!peerConnected && <p className="warning">Connect to Link first</p>}
        </div>
      </div>
      
      <div className="file-list">
        <h3>Transfer Queue</h3>
        {files.map((file, index) => (
          <div key={index} className="file-item">
            <span className="file-icon">{getFileIcon(file.name)}</span>
            <span className="file-name">{file.name}</span>
            <span className="file-status">{file.status}</span>
            {file.status === 'sending' && (
              <progress value={file.progress || 0} max="100" />
            )}
          </div>
        ))}
        {files.length === 0 && (
          <p className="empty-state">No files in transfer queue</p>
        )}
      </div>
    </div>
  );
};

// src/components/RemoteControlPanel.jsx
import React, { useState, useEffect, useRef } from 'react';
import './RemoteControlPanel.css';

const RemoteControlPanel = ({ remoteStream, peerConnected, onControlCommand }) => {
  const [controlMode, setControlMode] = useState('mouse'); // mouse, keyboard, gestures
  const [mousePosition, setMousePosition] = useState({ x: 0, y: 0 });
  const [isDrawing, setIsDrawing] = useState(false);
  const canvasRef = useRef(null);
  const remoteVideoRef = useRef(null);

  useEffect(() => {
    if (remoteStream && remoteVideoRef.current) {
      remoteVideoRef.current.srcObject = remoteStream;
    }
  }, [remoteStream]);

  const handleMouseMove = (e) => {
    if (!peerConnected) return;
    
    const rect = e.currentTarget.getBoundingClientRect();
    const x = ((e.clientX - rect.left) / rect.width) * 100;
    const y = ((e.clientY - rect.top) / rect.height) * 100;
    
    setMousePosition({ x, y });
    
    if (controlMode === 'mouse') {
      onControlCommand({
        type: 'mouse-move',
        x: x,
        y: y
      });
    }
  };

  const handleMouseClick = (e) => {
    if (!peerConnected) return;
    
    onControlCommand({
      type: 'mouse-click',
      button: e.button,
      x: mousePosition.x,
      y: mousePosition.y
    });
  };

  const handleKeyDown = (e) => {
    if (!peerConnected || controlMode !== 'keyboard') return;
    
    onControlCommand({
      type: 'keyboard',
      key: e.key,
      code: e.code,
      altKey: e.altKey,
      ctrlKey: e.ctrlKey,
      shiftKey: e.shiftKey
    });
  };

  const handleTouchStart = (e) => {
    if (!peerConnected || controlMode !== 'gestures') return;
    
    const touch = e.touches[0];
    const rect = e.currentTarget.getBoundingClientRect();
    const x = ((touch.clientX - rect.left) / rect.width) * 100;
    const y = ((touch.clientY - rect.top) / rect.height) * 100;
    
    setIsDrawing(true);
    
    onControlCommand({
      type: 'touch-start',
      x: x,
      y: y,
      force: touch.force || 0.5
    });
  };

  const handleTouchMove = (e) => {
    if (!peerConnected || controlMode !== 'gestures' || !isDrawing) return;
    e.preventDefault();
    
    const touch = e.touches[0];
    const rect = e.currentTarget.getBoundingClientRect();
    const x = ((touch.clientX - rect.left) / rect.width) * 100;
    const y = ((touch.clientY - rect.top) / rect.height) * 100;
    
    onControlCommand({
      type: 'touch-move',
      x: x,
      y: y
    });
  };

  const handleTouchEnd = () => {
    setIsDrawing(false);
    onControlCommand({ type: 'touch-end' });
  };

  return (
    <div className="remote-control-panel">
      <div className="control-header">
        <h3>Remote Control Interface</h3>
        <div className="mode-selector">
          <button 
            className={controlMode === 'mouse' ? 'active' : ''}
            onClick={() => setControlMode('mouse')}
          >🖱️ Mouse</button>
          <button 
            className={controlMode === 'keyboard' ? 'active' : ''}
            onClick={() => setControlMode('keyboard')}
          >⌨️ Keyboard</button>
          <button 
            className={controlMode === 'gestures' ? 'active' : ''}
            onClick={() => setControlMode('gestures')}
          >👆 Gestures</button>
        </div>
      </div>
      
      <div className="remote-screen">
        {remoteStream ? (
          <video 
            ref={remoteVideoRef}
            autoPlay
            playsInline
            className="remote-video"
          />
        ) : (
          <div className="remote-placeholder">
            <div className="placeholder-content">
              <span className="icon">📱</span>
              <p>Waiting for Link to share screen...</p>
              {!peerConnected && (
                <p className="hint">Connect to Link to enable remote control</p>
              )}
            </div>
          </div>
        )}
        
        {/* Overlay for mouse/touch interaction */}
        {peerConnected && remoteStream && (
          <div 
            className="interaction-overlay"
            onMouseMove={handleMouseMove}
            onClick={handleMouseClick}
            onTouchStart={handleTouchStart}
            onTouchMove={handleTouchMove}
            onTouchEnd={handleTouchEnd}
            tabIndex={0}
            onKeyDown={handleKeyDown}
          >
            {controlMode === 'mouse' && (
              <div 
                className="mouse-cursor"
                style={{ left: `${mousePosition.x}%`, top: `${mousePosition.y}%` }}
              >
                🖱️
              </div>
            )}
          </div>
        )}
      </div>
      
      <div className="control-tips">
        <h4>Current Mode: {controlMode}</h4>
        {controlMode === 'mouse' && (
          <p>Move mouse to control cursor, click to tap</p>
        )}
        {controlMode === 'keyboard' && (
          <p>Click on the screen and start typing</p>
        )}
        {controlMode === 'gestures' && (
          <p>Touch and drag on screen to draw gestures</p>
        )}
      </div>
    </div>

// src/components/ScreenShare.jsx
import React, { useEffect, useRef } from 'react';

const ScreenShare = ({ onStream }) => {
  const videoRef = useRef(null);

  useEffect(() => {
    let stream = null;
    
    const startScreenShare = async () => {
      try {
        // Request screen capture
        stream = await navigator.mediaDevices.getDisplayMedia({
          video: {
            cursor: "always"
          },
          audio: false
        });
        
        if (videoRef.current) {
          videoRef.current.srcObject = stream;
        }
        
        // Pass stream to parent for sending via Skylink
        onStream(stream);
        
        // Handle when user stops sharing
        stream.getVideoTracks()[0].onended = () => {
          console.log('Screen sharing ended');
          onStream(null);
        };
        
      } catch (err) {
        console.error('Error sharing screen:', err);
      }
    };
    
    startScreenShare();
    
    return () => {
      if (stream) {
        stream.getTracks().forEach(track => track.stop());
      }
    };
  }, [onStream]);

  return (
    <div className="screen-share">
      <video 
        ref={videoRef}
        autoPlay 
        playsInline
        className="screen-video"
      />
    </div>
  );
};

Important Considerations & Limitations
Web-Based Remote Control Limitations
Important: Full remote device control (like tapping apps, accessing system functions) requires a native mobile application paired with your web interface. The web version demonstrated here provides:

Screen sharing from Link to Sky

Mouse/keyboard simulation within the browser context

Basic gesture recognition

For true device control (similar to TeamViewer), you'll need to build companion native apps using:

Android: Skylink Android SDK 

iOS: Skylink iOS SDK 

React Native: Skylink React Native SDK for cross-platform 

Privacy & Security
All communication is peer-to-peer and encrypted via WebRTC

No messages or files pass through your servers

The signaling server only helps establish the connection 

Users must explicitly share their screen - cannot be forced remotely

Performance Optimization
For large file transfers, implement chunking and resume capability

Consider bandwidth limitations - WebRTC adapts to network conditions

For production, implement reconnection logic and error boundaries

Summary
This SkyLink implementation provides a complete foundation for a dual-interface peer-to-peer communication system. The key achievements are:

Dual Interface Architecture: Separate "Sky" (controller) and "Link" (device) experiences with shared underlying connection logic

Real-time Communication: Powered by SkylinkJS WebRTC SDK for low-latency peer-to-peer data transfer 

Multi-modal Interaction: Chat, file transfer, and remote control capabilities

Screen Sharing: Link devices can broadcast their screen to Sky controllers

Extensible Design: Clear separation of concerns makes adding new features straightforward

The system demonstrates how two web clients can form a direct "SkyLink" connection, enabling the kind of device interaction seen in WhatsApp Web while adding advanced features like file transfer and screen control. For full mobile device control, extend this foundation with the native Skylink SDKs for Android and iOS .
